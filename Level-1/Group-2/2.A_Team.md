import java.util.Scanner;

public class A_Team {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();

        int count = 0;
        for (int i = 0; i < n; i++) {
            int tmpCount = 0;
            for (int j = 0; j < 3; j++) {
                if (sc.nextInt() == 1)
                    tmpCount++;
            }
            if (tmpCount >= 2)
                count++;
        }
        System.out.println(count);
    }
}
